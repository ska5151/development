```bash
@echo off

 :: 관리자 모드 실행
 :-------------------------------------
 REM  --> Check for permissions
 >nul 2>&1 "%SYSTEMROOT%\system32\cacls.exe" "%SYSTEMROOT%\system32\config\system"

REM --> If error flag set, we do not have admin.
 if '%errorlevel%' NEQ '0' (
     echo Requesting administrative privileges...
     goto UACPrompt
 ) else ( goto gotAdmin )

:UACPrompt
     echo Set UAC = CreateObject^("Shell.Application"^) > "%temp%\getadmin.vbs"
     echo UAC.ShellExecute "%~s0", "", "", "runas", 1 >> "%temp%\getadmin.vbs"

    "%temp%\getadmin.vbs"
     exit /B

:gotAdmin
     if exist "%temp%\getadmin.vbs" ( del "%temp%\getadmin.vbs" )
     pushd "%CD%"
     CD /D "%~dp0"
 :--------------------------------------

:: 현재 날짜와 시간 가져오기
set "Ymd=%date:~0,4%-%date:~5,2%-%date:~8,2%"

set "log_name_Ap=Application"
set "output_file=%userprofile%\Desktop\로그저장\%Ymd%-응용.evtx"
set "output_file_txt=%userprofile%\Desktop\로그저장\%Ymd%-응용.log"

echo %log_name_Ap% Application 로그를 내보내는 중...
wevtutil epl "%log_name_Ap%" "%output_file%"
wevtutil qe %log_name_Ap% /f:text > "%output_file_txt%"
echo %log_name_Ap% Application 로그 내보내기 완료:

set "log_name_Sy=System"
set "output_file=%userprofile%\Desktop\로그저장\%Ymd%-시스템.evtx"
set "output_file_txt=%userprofile%\Desktop\로그저장\%Ymd%-시스템.log"

echo %log_name_Sy% System 로그를 내보내는 중...
wevtutil epl "%log_name_Sy%" "%output_file%"
wevtutil qe %log_name_Sy% /f:text > "%output_file_txt%"
echo %log_name_Sy% System 로그 내보내기 완료:

set "log_name_Se=Security"
set "output_file=%userprofile%\Desktop\로그저장\%Ymd%-보안.evtx"
set "output_file_txt=%userprofile%\Desktop\로그저장\%Ymd%-보안.log"

echo %log_name_Se% Security 로그를 내보내는 중...
wevtutil epl "%log_name_Se%" "%output_file%"
wevtutil qe %log_name_Se% /f:text > "%output_file_txt%"
echo %log_name_Se% Security 로그 내보내기 완료:

echo %log_name_Ap% 로그를 삭제하는 중...
wevtutil cl %log_name_Ap%

echo %log_name_Sy% 로그를 삭제하는 중...
wevtutil cl %log_name_Sy%

echo %log_name_Se% 로그를 삭제하는 중...
wevtutil cl %log_name_Se%
```
